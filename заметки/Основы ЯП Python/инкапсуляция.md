```Python 
class Person:

    def __init__(self, name):

        self.__name = name  # "приватное" поле

  

    def get_name(self):

        return self.__name

  

    def set_name(self, name):

        if isinstance(name, str) and name:

            self.__name = name

  

    name = property(get_name, set_name)

  

p = Person("Иван")

print(p.name)    # вывод: Иван

p.name = "Пётр"

print(p.name)    # вывод: Пётр

  

# Прямой доступ к __name затруднён, но возможен через манглинг имени:

# print(p.__name)    # вызовет ошибку

print(p._Person__name) # выводит: Пётр
```
