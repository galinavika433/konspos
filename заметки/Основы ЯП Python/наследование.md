```Python
# Базовый класс
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "Неизвестный звук"

# Класс-наследник
class Dog(Animal):
    def speak(self):
        return "Гав"

class Cat(Animal):
    def speak(self):
        return "Мяу"

dog = Dog("Бобик")
cat = Cat("Мурка")

print(dog.name, "говорит:", dog.speak())   # Бобик говорит: Гав
print(cat.name, "говорит:", cat.speak())   # Мурка говорит: Мяу
```
